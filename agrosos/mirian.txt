# --- Fase 1: Construir la aplicación ---

FROM node:22-alpine AS builder

WORKDIR /usr/src/app/backend

COPY ../package.json ../package-lock.json* ./

# Limpia la caché de npm antes de instalar
RUN npm cache clean --force

RUN npm install

COPY . .

# Elimina la caché de Vite
RUN rm -rf node_modules/.vite

RUN npm run build

# --- Fase 2: Crear la imagen de producción ---

FROM node:22-alpine

WORKDIR /usr/src/app

COPY --from=builder /usr/src/app/backend/dist ./dist
COPY --from=builder /usr/src/app/backend/node_modules ./node_modules
COPY --from=builder /usr/src/app/backend/package.json ./package.json
COPY --from=builder /usr/src/app/backend/express.js ./express.js

EXPOSE 3000

CMD ["npm", "start"]
